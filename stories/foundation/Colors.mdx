# Colors

Design tokens use **RGB channel values** so opacity can be composed at the call site:

```css
rgb(var(--accent))         /* full opacity */
rgb(var(--accent) / 0.5)   /* 50% opacity */
```

Swatches update when you switch the Storybook theme.

export const Pair = ({ fill, fg, label, note }) => (
  <div
    style={{
      display: 'inline-flex',
      alignItems: 'center',
      gap: 10,
      borderRadius: 8,
      border: '1px solid rgb(var(--border))',
      padding: '8px 12px',
      background: `rgb(var(${fill}))`,
    }}
  >
    <span style={{ color: fg ? `rgb(var(${fg}))` : 'rgb(var(--foreground))', fontWeight: 600, fontSize: 14 }}>Aa</span>
    <div style={{ display: 'flex', flexDirection: 'column', gap: 2 }}>
      <code style={{ fontSize: 12, color: fg ? `rgb(var(${fg}))` : 'rgb(var(--foreground))' }}>{fill}</code>
      {fg && <code style={{ fontSize: 11, color: fg ? `rgb(var(${fg}) / 0.7)` : 'rgb(var(--muted-foreground))' }}>{fg}</code>}
    </div>
    {note && <span style={{ fontSize: 11, color: fg ? `rgb(var(${fg}) / 0.6)` : 'rgb(var(--muted-foreground))', marginLeft: 4 }}>{note}</span>}
  </div>
)

export const Solo = ({ token, label, note }) => (
  <div style={{ display: 'flex', alignItems: 'center', gap: 12, padding: '4px 0' }}>
    <div
      style={{
        width: 36,
        height: 36,
        borderRadius: 6,
        flexShrink: 0,
        background: `rgb(var(${token}))`,
        border: '1px solid rgb(var(--border))',
      }}
    />
    <div>
      <code style={{ fontWeight: 600, fontSize: 13 }}>{token}</code>
      <span style={{ marginLeft: 10, fontSize: 13, color: 'rgb(var(--muted-foreground))' }}>{label}</span>
      {note && <span style={{ marginLeft: 8, fontSize: 11, color: 'rgb(var(--muted-foreground))', opacity: 0.75 }}>{note}</span>}
    </div>
  </div>
)
